// Function to display browser information
function displayBrowserInfo() {
    // Accessing browser information using navigator object
    const browserInfo = {
        appName: navigator.appName,
        appVersion: navigator.appVersion,
        userAgent: navigator.userAgent,
        platform: navigator.platform,
        language: navigator.language,
        cookiesEnabled: navigator.cookieEnabled,
        onlineStatus: navigator.onLine ? 'Online' : 'Offline'
    };

    // Displaying browser information
    console.log('Browser Information:');
    console.log('---------------------');
    console.log('Name:', browserInfo.appName);
    console.log('Version:', browserInfo.appVersion);
    console.log('User Agent:', browserInfo.userAgent);
    console.log('Platform:', browserInfo.platform);
    console.log('Language:', browserInfo.language);
    console.log('Cookies Enabled:', browserInfo.cookiesEnabled);
    console.log('Online Status:', browserInfo.onlineStatus);
}

// Call the function to display browser information
displayBrowserInfo();
